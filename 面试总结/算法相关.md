function flatten_deep(arr, deep) {
    let result = [];
    if (Array.isArray(arr) && deep > 0) {
        for (let i = 0; i < arr.length; i++) {
            result = result.concat(flatten_deep(arr[i], deep - 1));
        }
    }
    else {
        result.push(arr);
    }
    return result;
}

console.log(flatten_deep(["a", ["b", "c", [7, 8]],[4,5,[6,["dd","whidhw"]]]],3));
